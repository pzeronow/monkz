def create_sine(
    steps,
    length=1,
    vertical_shift=1,
    amplitude=1,
    cycles=2
):
    t = np.arange(0.0, length, steps)
    s = vertical_shift*amplitude + np.sin(cycles * np.pi * t)
    return s

def add_noise_to(v, noise_amplitude):
    noise = np.random.rand(1, len(v))
    t = np.ones(len(v))
    a_t = np.dot(noise_amplitude, t)
    n_a = noise ** a_t
    return np.add(v, n_a)[0]
